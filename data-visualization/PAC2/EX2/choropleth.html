import pandas as pd
import folium

# Load the data 
data = pd.read_csv('./data/statedata.csv')

# Convert the 'Life.Exp' column to numeric
data['Life.Exp'] = pd.to_numeric(data['Life.Exp'], errors='coerce')

world_geo = './data/us-states.json'

# Create a map centered at [0, 0] and zoom level 2
m = folium.Map(location=[37, -102], zoom_start=3)

# Add the color for the chloropleth:
choropleth = folium.Choropleth(
    geo_data=world_geo,
    data=data,
    columns=['state.name', 'Life.Exp'],
    key_on='feature.properties.name',
    fill_color='YlGn',
    fill_opacity=0.7,
    line_opacity=0.2,
    legend_name='Life expectancy in years',
    smooth_factor=0
).add_to(m)


folium.GeoJsonTooltip(fields=["name"]).add_to(choropleth.geojson)
folium.LayerControl().add_to(m)
